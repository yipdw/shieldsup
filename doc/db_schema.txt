create table tokens (
    id int unsigned not null auto_increment primary key,
    userid bigint not null,
    oauth_token text,
    oauth_secret text,
    accessed timestamp,
    added timestamp
);

create table jobs (
    id int unsigned not null auto_increment primary key,
    userid bigint not null,
    target_userid bigint,
    target_username varchar(24),
    opt_rt boolean default 1,
    opt_reply boolean default 0,
    opt_filter_friends boolean default 1,
    updated timestamp,
    added timestamp,
    status varchar(10),
    errcode varchar(20),
    outputguid varchar(36)
);

create table blobs (
    id int unsigned not null auto_increment primary key,
    userid bigint not null,
    guid varchar(36),
    filename varchar(255),
    data blob
);
